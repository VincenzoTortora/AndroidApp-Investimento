package com.vincenzo.borsaapp

import android.content.Intent
import android.os.Bundle
import android.view.View
import android.widget.Button
import android.widget.EditText
import android.widget.TextView
import android.widget.Toast
import androidx.appcompat.app.AlertDialog
import androidx.appcompat.app.AppCompatActivity

class CalcolaDividendiActivity : AppCompatActivity() {
    private lateinit var dbHelper: DatabaseHelper
    private lateinit var etImporto: EditText
    private lateinit var btnCalcolaDividendi: Button
    private lateinit var btnSommaCapitale: Button
    private lateinit var txtRisultato: TextView
    private var dividendiNetti: Double = 0.0

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_calcola_dividendi)

        dbHelper = DatabaseHelper(this)
        etImporto = findViewById(R.id.etImportoDividendi)
        btnCalcolaDividendi = findViewById(R.id.btnCalcolaDividendi)
        btnSommaCapitale = findViewById(R.id.btnSommaAlCapitale)
        txtRisultato = findViewById(R.id.txtRisultato)

        btnCalcolaDividendi.setOnClickListener { calcolaDividendi() }
        btnSommaCapitale.setOnClickListener { mostraConfermaSommaAlCapitale() }
    }

    private fun calcolaDividendi() {
        val importo = etImporto.text.toString().toDoubleOrNull()
        if (importo != null && importo > 0) {
            dividendiNetti = importo * 0.74 // Calcola il dividendo netto (26% di tasse)
            txtRisultato.text = "Dividendi netti: € ${"%.2f".format(dividendiNetti)}"
        } else {
            Toast.makeText(this, "Inserire un importo valido", Toast.LENGTH_SHORT).show()
        }
    }

    private fun mostraConfermaSommaAlCapitale() {
        if (dividendiNetti > 0) {
            AlertDialog.Builder(this)
                .setTitle("Conferma aggiunta al capitale")
                .setMessage("Sei sicuro di voler sommare € ${"%.2f".format(dividendiNetti)} al capitale?")
                .setPositiveButton("Conferma") { _, _ -> sommaAlCapitale() }
                .setNegativeButton("Annulla") { dialog, _ -> dialog.dismiss() }
                .setCancelable(false)
                .show()
        } else {
            Toast.makeText(this, "Nessun dividendo valido da sommare", Toast.LENGTH_SHORT).show()
        }
    }

    private fun sommaAlCapitale() {
        if (dividendiNetti > 0) {
            val capitaleAttuale = dbHelper.readCapitaleIniziale()
            val nuovoCapitale = capitaleAttuale + dividendiNetti

            // Aggiorna il capitale nel database
            dbHelper.updateCapitaleIniziale(nuovoCapitale)

            // Mostra il messaggio di conferma
            Toast.makeText(this, "Capitale aggiornato: € ${"%.2f".format(nuovoCapitale)}", Toast.LENGTH_SHORT).show()

            // Torna alla MainActivity e passa il capitale aggiornato
            val intent = Intent(this, MainActivity::class.java)
            intent.putExtra("CAPITALE_AGGIORNATO", nuovoCapitale)
            startActivity(intent)
        } else {
            Toast.makeText(this, "Nessun dividendo valido da sommare", Toast.LENGTH_SHORT).show()
        }
    }


}
